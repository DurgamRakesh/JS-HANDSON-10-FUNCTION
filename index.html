Question 7 : Write a function that accepts parameter n and returns factorial of n 

        const myfac = (n) => {
    
            if(n == 0){
                return 1;
            }
            else{
                return n * myfac(n-1);
            }
            
        }
        const factorial = myfac(5);
        console.log(factorial);

Output: 120
